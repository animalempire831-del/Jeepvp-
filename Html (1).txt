    let users = JSON.parse(localStorage.getItem('vp_users')) || [];
    let currentUser = JSON.parse(sessionStorage.getItem('vp_current')) || null;

    function saveAll() {
        localStorage.setItem('vp_users', JSON.stringify(users));
        localStorage.setItem('vp_deposits', JSON.stringify(deposits));
        localStorage.setItem('vp_withdrawals', JSON.stringify(withdrawals));
        localStorage.setItem('vp_investments', JSON.stringify(investments));
        localStorage.setItem('vp_notifications', JSON.stringify(notifications));
    }

    function addNotification(msg, type="info") {
        notifications.unshift({ msg, date: new Date().toLocaleString(), type });
        if(notifications.length>20) notifications.pop();
        saveAll();
        showToast(msg);
    }

    function showToast(msg) {
        let toastDiv = document.createElement('div');
        toastDiv.className = 'toast-notify';
        toastDiv.innerHTML = `<i class="fas fa-bell text-gold-400 mr-2"></i> ${msg}`;
        document.body.appendChild(toastDiv);
        setTimeout(()=> toastDiv.remove(), 3500);
    }

    function updateBalanceUI() { 
        if(currentUser) {
            let balSpan = document.getElementById('liveBalanceVal');
            if(balSpan) balSpan.innerText = currentUser.balance.toLocaleString();
            let earningSpan = document.getElementById('dailyEarningSpan');
            if(earningSpan) earningSpan.innerText = currentUser.dailyProfit?.toFixed(0) || 0;
        }
    }

    function calcDailyProfit(user) {
        let totalDaily = 0;
        investments.forEach(inv => { if(inv.userId === user.id && inv.status === 'active') totalDaily += inv.dailyRate; });
        return totalDaily;
    }

    function updateUserDaily() {
        if(currentUser) {
            let dp = calcDailyProfit(currentUser);
            currentUser.dailyProfit = dp;
            let idx = users.findIndex(u=> u.id === currentUser.id);
            if(idx!==-1) users[idx].dailyProfit = dp;
            saveAll();
            updateBalanceUI();
        }
    }

    // auto daily cron simulation (each minute)
    setInterval(() => {
        if(currentUser) {
            let now = new Date();
            let lastClaim = currentUser.lastProfitClaim ? new Date(currentUser.lastProfitClaim) : null;
            if(!lastClaim || (now - lastClaim) >= 86400000) {
                let dailyAdd = currentUser.dailyProfit || 0;
                if(dailyAdd>0){
                    currentUser.balance += dailyAdd;
                    currentUser.lastProfitClaim = now.toISOString();
                    let idx = users.findIndex(u=> u.id===currentUser.id);
                    if(idx!==-1) users[idx].balance = currentUser.balance;
                    saveAll();
                    addNotification(`✨ +${dailyAdd.toLocaleString()} RWF daily profit credited!`, "profit");
                    updateBalanceUI();
                    renderCurrentPage();
                }
            }
        }
    }, 60000);

    function checkInvestmentCompletion() {
        investments.forEach(inv => {
            if(inv.status === 'active' && inv.endDate && new Date(inv.endDate) <= new Date()){
                inv.status = 'completed';
                addNotification(`✅ Plan ${inv.planName} completed. Total earnings finished.`, "success");
            }
        });
        saveAll();
    }
    setInterval(checkInvestmentCompletion, 30000);

        let pageId = sessionStorage.getItem('currentPage') || 'landing';

        if(target) target.classList.remove('hidden');
        if(pageId === 'dashboardPage') renderDashboard();
        else if(pageId === 'depositPage') renderDeposit();
        else if(pageId === 'withdrawalPage') renderWithdrawal();
        else if(pageId === 'incomePlansPage') renderIncomePlans();
        else if(pageId === 'historyPage') renderHistory();
        else if(pageId === 'profilePage') renderProfile();
        else if(pageId === 'supportPage') renderSupport();
        else if(pageId === 'adminPanel') renderAdmin();
        else if(pageId === 'signupPage') renderSignup();
        else if(pageId === 'loginPage') renderLogin();
        else renderLanding();
    }
